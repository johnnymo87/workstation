# LGTM v2 Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Rewrite lgtm as a context-aware PR review daemon for cloudbox -- a thin discovery + context pipeline + dispatch layer that builds a deterministic context packet and sends grounded reviews to OpenCode headless sessions.

**Architecture:** Systemd timer triggers a oneshot service every 10 minutes. The service discovers PRs via `gh`, builds a context packet (PR history, repo rules, author activity, session history), creates worktrees for full codebase access, dispatches reviews to OpenCode via `opencode-launch`, and tracks outcomes with JSON files.

**Tech Stack:** TypeScript (ESM, tsx runner), Node.js, gh CLI, git, opencode-launch, oc-search, vitest

**Existing code:** The `~/projects/lgtm` repo has a working macOS-oriented v1. We reuse `github.ts` (PR discovery) and `worktree.ts` patterns, but replace the macOS-specific parts (launchd, SQLite, node-notifier, commander CLI, interactive review mode) with cloudbox patterns (flat-file state, opencode-launch, systemd timer).

**Design doc:** `docs/plans/2026-04-16-lgtm-v2-design.md` in the workstation repo.

```typescript
import { existsSync, mkdirSync, writeFileSync, readFileSync } from "node:fs";
import { join } from "node:path";
import type { OutcomeRecord } from "./types.js";

/**
 * Check if a PR has already been dispatched for review.
 * State: <stateDir>/dispatched/<org>/<repo>/<pr-number>
 */
export function isDispatched(stateDir: string, repoFullName: string, prNumber: number): boolean {
  return existsSync(getMarkerPath(stateDir, repoFullName, prNumber));
}

/**
 * Mark a PR as dispatched by creating a marker file.
 */
export function markDispatched(stateDir: string, repoFullName: string, prNumber: number): void {
  const markerPath = getMarkerPath(stateDir, repoFullName, prNumber);
  const dir = join(markerPath, "..");
  mkdirSync(dir, { recursive: true });
  writeFileSync(markerPath, "");
}

function getMarkerPath(stateDir: string, repoFullName: string, prNumber: number): string {
  return join(stateDir, "dispatched", repoFullName, String(prNumber));
}

/**
 * Write an outcome record for a dispatched review.
 */
export function writeOutcome(stateDir: string, outcome: OutcomeRecord): void {
  const outcomePath = join(stateDir, "outcomes", outcome.repo, `${outcome.prNumber}.json`);
  const dir = join(outcomePath, "..");
  mkdirSync(dir, { recursive: true });
  writeFileSync(outcomePath, JSON.stringify(outcome, null, 2));
}

/**
 * Read an outcome record.
 */
export function readOutcome(stateDir: string, repo: string, prNumber: number): OutcomeRecord | null {
  const outcomePath = join(stateDir, "outcomes", repo, `${prNumber}.json`);
  if (!existsSync(outcomePath)) return null;
  return JSON.parse(readFileSync(outcomePath, "utf-8"));
}
```

```typescript
import { execFile } from "node:child_process";
import { promisify } from "node:util";
import type { LgtmConfig, PrInfo, PrDetail } from "./types.js";

const execFileAsync = promisify(execFile);

/**
 * Discover open PRs where the current user is requested as a reviewer.
 */
export async function discoverPrs(config: LgtmConfig): Promise<PrInfo[]> {
  const { stdout } = await execFileAsync("gh", [
    "search", "prs",
    "user-review-requested:@me",
    "--owner", config.org,
    "--state", "open",
    "--json", "repository,number,title,author",
  ]);

  const raw = JSON.parse(stdout);
  return parsePrSearchResult(raw, []);
}

/**
 * Parse gh search output into PrInfo array. Exported for testing.
 */
export function parsePrSearchResult(
  raw: Array<{ repository: { nameWithOwner: string }; number: number; title: string; author: { login: string } }>,
  excludeRepos: string[],
): PrInfo[] {
  return raw
    .filter((pr) => !excludeRepos.includes(pr.repository.nameWithOwner))
    .map((pr) => ({
      repo: pr.repository.nameWithOwner,
      number: pr.number,
      title: pr.title,
      author: pr.author.login,
    }));
}

/**
 * Get detailed PR info including diff and changed file list.
 */
export async function getPrDetail(pr: PrInfo): Promise<PrDetail> {
  const [metaResult, diffResult] = await Promise.all([
    execFileAsync("gh", [
      "pr", "view", String(pr.number),
      "--repo", pr.repo,
      "--json", "additions,deletions,changedFiles,body,labels,headRefName,files",
    ]),
    execFileAsync("gh", [
      "pr", "diff", String(pr.number),
      "--repo", pr.repo,
    ]),
  ]);

  const meta = JSON.parse(metaResult.stdout);
  return parsePrDetail(meta, diffResult.stdout);
}

/**
 * Parse gh pr view JSON + diff into PrDetail. Exported for testing.
 */
export function parsePrDetail(
  meta: {
    additions: number;
    deletions: number;
    changedFiles: number;
    body?: string;
    labels?: Array<{ name: string }>;
    headRefName: string;
    files?: Array<{ path: string }>;
  },
  diff: string,
): PrDetail {
  return {
    additions: meta.additions,
    deletions: meta.deletions,
    changedFiles: meta.changedFiles,
    diff,
    body: meta.body ?? "",
    labels: (meta.labels ?? []).map((l) => l.name),
    headRefName: meta.headRefName,
    files: (meta.files ?? []).map((f) => f.path),
  };
}
```

```typescript
import { execFile } from "node:child_process";
import { promisify } from "node:util";
import { existsSync, mkdirSync } from "node:fs";
import { join } from "node:path";
import { log } from "./log.js";

const execFileAsync = promisify(execFile);

/**
 * Find the local clone path for a repo in the projects directory.
 */
export function resolveRepoPath(projectsDir: string, repoFullName: string): string | null {
  const repoName = repoFullName.split("/").pop()!;
  const candidate = join(projectsDir, repoName);
  if (existsSync(join(candidate, ".git"))) {
    return candidate;
  }
  return null;
}

/**
 * Clone a repo if it doesn't exist locally.
 */
export async function ensureRepo(
  projectsDir: string,
  repoFullName: string,
): Promise<string> {
  const existing = resolveRepoPath(projectsDir, repoFullName);
  if (existing) return existing;

  const repoName = repoFullName.split("/").pop()!;
  const targetPath = join(projectsDir, repoName);

  log(`Cloning ${repoFullName} to ${targetPath}`);
  await execFileAsync("git", [
    "clone", `git@github.com:${repoFullName}.git`, targetPath,
  ], { timeout: 120_000 });

  return targetPath;
}

/**
 * Create a git worktree for a PR branch.
 * Worktrees go in <repo>/.worktrees/pr-<N>.
 */
export async function createWorktree(
  repoPath: string,
  prNumber: number,
  headRefName: string,
): Promise<string | null> {
  const worktreeDir = join(repoPath, ".worktrees");
  if (!existsSync(worktreeDir)) {
    mkdirSync(worktreeDir, { recursive: true });
  }
  const worktreePath = join(worktreeDir, `pr-${prNumber}`);

  try {
    await execFileAsync("git", ["fetch", "origin"], {
      cwd: repoPath, timeout: 60_000,
    });

    if (existsSync(worktreePath)) {
      await execFileAsync("git", ["worktree", "remove", worktreePath, "--force"], {
        cwd: repoPath, timeout: 15_000,
      }).catch(() => {});
    }

    await execFileAsync("git", [
      "worktree", "add", worktreePath,
      `origin/${headRefName}`, "--detach",
    ], { cwd: repoPath, timeout: 30_000 });

    log(`Created worktree at ${worktreePath}`);
    return worktreePath;
  } catch (err) {
    log(`Failed to create worktree for PR #${prNumber}: ${err}`);
    return null;
  }
}

/**
 * Remove a worktree.
 */
export async function removeWorktree(repoPath: string, worktreePath: string): Promise<void> {
  try {
    await execFileAsync("git", ["worktree", "remove", worktreePath, "--force"], {
      cwd: repoPath, timeout: 15_000,
    });
  } catch {
    // Best-effort cleanup
  }
}
```

```typescript
import { execFile } from "node:child_process";
import { promisify } from "node:util";

const execFileAsync = promisify(execFile);

/**
 * Run a shell command and return stdout, or empty string on failure.
 * Failures are swallowed because context building is best-effort.
 */
export async function run(cmd: string, args: string[], opts?: { cwd?: string; timeout?: number }): Promise<string> {
  try {
    const { stdout } = await execFileAsync(cmd, args, {
      timeout: opts?.timeout ?? 30_000,
      maxBuffer: 10 * 1024 * 1024,
      cwd: opts?.cwd,
    });
    return stdout;
  } catch {
    return "";
  }
}
```

**Step 4: Write context builder implementation**

Create `src/context.ts`:
```typescript
import { readFileSync, existsSync } from "node:fs";
import { join, dirname } from "node:path";
import { run } from "./shell.js";
import { log } from "./log.js";
import type { PrInfo, PrDetail, LgtmConfig } from "./types.js";

const CONTEXT_CHAR_LIMIT = 40_000; // ~10K tokens

/**
 * Build the full context packet as a markdown string.
 * Each section is best-effort -- failures in one section don't block others.
 */
export async function buildContextPacket(
  pr: PrInfo,
  detail: PrDetail,
  config: LgtmConfig,
  worktreePath: string,
): Promise<string> {
  const sections: string[] = [];

  // 1. PR Metadata
  sections.push(buildPrMetadataSection(pr, detail));

  // 2. Repo Rules
  sections.push(buildRepoRulesSection(worktreePath));

  // 3. Changed File Clusters
  sections.push(buildFileClustersSection(detail.files));

  // 4-7 are async -- run in parallel for speed
  const [recentMerged, authorActivity, reviewComments, sessionHits] = await Promise.all([
    buildRecentMergedSection(pr, detail),
    buildAuthorActivitySection(pr),
    buildReviewCommentsSection(pr, detail),
    buildSessionHistorySection(pr, detail),
  ]);

  sections.push(recentMerged);
  sections.push(authorActivity);
  sections.push(reviewComments);
  sections.push(sessionHits);

  const packet = sections.filter(Boolean).join("\n\n---\n\n");
  return truncateToCharLimit(packet, CONTEXT_CHAR_LIMIT);
}

function buildPrMetadataSection(pr: PrInfo, detail: PrDetail): string {
  return `## PR Metadata

- **Repository:** ${pr.repo}
- **PR #${pr.number}:** ${pr.title}
- **Author:** ${pr.author}
- **Branch:** ${detail.headRefName}
- **Labels:** ${detail.labels.join(", ") || "(none)"}
- **Stats:** +${detail.additions} -${detail.deletions} across ${detail.changedFiles} files
- **Description:** ${detail.body || "(no description)"}`;
}

function buildRepoRulesSection(worktreePath: string): string {
  const rulesFiles = ["AGENTS.md", "CLAUDE.md", ".github/review-guide.md", ".github/CODEOWNERS"];
  const found: string[] = [];

  for (const file of rulesFiles) {
    const fullPath = join(worktreePath, file);
    if (existsSync(fullPath)) {
      try {
        const content = readFileSync(fullPath, "utf-8");
        // Cap each rules file at 3000 chars
        const truncated = content.length > 3000
          ? content.slice(0, 3000) + "\n\n[... truncated]"
          : content;
        found.push(`### ${file}\n\n${truncated}`);
      } catch {
        // Skip unreadable files
      }
    }
  }

  if (found.length === 0) return "";
  return `## Repo Rules\n\n${found.join("\n\n")}`;
}

function buildFileClustersSection(files: string[]): string {
  if (files.length === 0) return "";
  const clusters = clusterFilesByDirectory(files);
  const lines = Object.entries(clusters).map(([dir, paths]) =>
    `- **${dir}/**: ${paths.map((p) => p.split("/").pop()).join(", ")}`
  );
  return `## Changed File Clusters\n\n${lines.join("\n")}`;
}

async function buildRecentMergedSection(pr: PrInfo, detail: PrDetail): Promise<string> {
  // Get recent merged PRs in this repo
  const raw = await run("gh", [
    "pr", "list",
    "--repo", pr.repo,
    "--state", "merged",
    "--limit", "15",
    "--json", "number,title,author,mergedAt,files",
  ]);

  if (!raw) return "";

  try {
    const prs = JSON.parse(raw) as Array<{
      number: number;
      title: string;
      author: { login: string };
      mergedAt: string;
      files?: Array<{ path: string }>;
    }>;

    // Filter to PRs touching the same paths
    const changedPaths = new Set(detail.files);
    const relevant = prs.filter((mergedPr) => {
      const mergedFiles = (mergedPr.files ?? []).map((f) => f.path);
      return mergedFiles.some((f) => changedPaths.has(f));
    }).slice(0, 5);

    if (relevant.length === 0) return "";

    const lines = relevant.map((p) => {
      const date = new Date(p.mergedAt).toISOString().split("T")[0];
      return `- #${p.number} (${date}, @${p.author.login}): ${p.title}`;
    });

    return `## Recent Merged PRs Touching Same Paths\n\n${lines.join("\n")}`;
  } catch {
    return "";
  }
}

async function buildReviewCommentsSection(pr: PrInfo, detail: PrDetail): Promise<string> {
  // Get recent merged PRs, then fetch review comments for ones touching same paths
  const raw = await run("gh", [
    "pr", "list",
    "--repo", pr.repo,
    "--state", "merged",
    "--limit", "10",
    "--json", "number,files",
  ]);

  if (!raw) return "";

  try {
    const prs = JSON.parse(raw) as Array<{
      number: number;
      files?: Array<{ path: string }>;
    }>;

    const changedPaths = new Set(detail.files);
    const relevantPrNumbers = prs
      .filter((p) => (p.files ?? []).some((f) => changedPaths.has(f.path)))
      .map((p) => p.number)
      .slice(0, 3); // Limit to 3 PRs to avoid too many API calls

    if (relevantPrNumbers.length === 0) return "";

    const comments: string[] = [];
    for (const prNum of relevantPrNumbers) {
      const reviewRaw = await run("gh", [
        "api", `repos/${pr.repo}/pulls/${prNum}/comments`,
        "--jq", '.[] | "- **\(.path):\(.line // .original_line)** (@\(.user.login)): \(.body | split("\n") | first)"',
      ]);
      if (reviewRaw.trim()) {
        comments.push(`### PR #${prNum}\n\n${reviewRaw.trim()}`);
      }
    }

    if (comments.length === 0) return "";
    return `## Past Review Comments on Similar Paths\n\n${comments.join("\n\n")}`;
  } catch {
    return "";
  }
}

async function buildSessionHistorySection(pr: PrInfo, detail: PrDetail): Promise<string> {
  const repoName = pr.repo.split("/").pop()!;

  // Search for the repo name in session history
  const raw = await run("oc-search", ["--types", "tool,text", repoName]);

  if (!raw.trim()) return "";

  // Take first 2000 chars of results
  const truncated = raw.length > 2000
    ? raw.slice(0, 2000) + "\n\n[... more results available via oc-search]"
    : raw;

  return `## Session History (oc-search "${repoName}")\n\n\`\`\`\n${truncated}\n\`\`\``;
}

/**
 * Group file paths by their parent directory (2 levels deep).
 * Exported for testing.
 */
export function clusterFilesByDirectory(files: string[]): Record<string, string[]> {
  const clusters: Record<string, string[]> = {};
  for (const file of files) {
    const parts = file.split("/");
    // Use first 2 path segments as the cluster key, or "." for root files
    const key = parts.length <= 1 ? "." : parts.slice(0, 2).join("/");
    if (!clusters[key]) clusters[key] = [];
    clusters[key].push(file);
  }
  return clusters;
}

/**
 * Truncate text to a character limit with a message.
 * Exported for testing.
 */
export function truncateToCharLimit(text: string, limit: number): string {
  if (text.length <= limit) return text;
  return text.slice(0, limit) + "\n\n[truncated -- context packet exceeded size limit]";
}
```

```typescript
import type { PrInfo, LgtmConfig } from "./types.js";

/**
 * Build the 3-phase review prompt for an OpenCode headless session.
 * The prompt assumes .lgtm-context.md is in the working directory.
 */
export function buildReviewPrompt(pr: PrInfo, config: LgtmConfig): string {
  const autoApproveNote = config.autoApproveAuthors.length > 0
    ? `Auto-approve eligible authors: ${config.autoApproveAuthors.join(", ")}. This PR's author is "${pr.author}".`
    : "No authors are eligible for auto-approve.";

  const sensitiveNote = config.sensitivePatterns.length > 0
    ? `Sensitive file patterns (never auto-approve): ${config.sensitivePatterns.join(", ")}`
    : "";

  return `You are reviewing pull request #${pr.number} in ${pr.repo}.
You have the full repository checked out at the PR branch, and a context
packet at .lgtm-context.md containing prior PR history, repo rules,
author activity, and session history.

## Phase 1: Ground Yourself

1. Read .lgtm-context.md thoroughly. It contains:
   - PR metadata and description
   - Repo rules (AGENTS.md, review guides, CODEOWNERS)
   - Recent merged PRs touching the same files
   - This author's recent merged PRs
   - Past review comments on similar paths
   - Relevant OpenCode session history
   - Changed file clusters by subsystem

2. Read the PR description and any linked issue.

3. Inspect the changed files and understand which subsystems are touched.

4. If the context packet references specific prior PRs or review comments
   that seem directly relevant, you may run \`gh pr view <N> --repo ${pr.repo}\`
   to read them in more detail.

5. If you need more context about prior discussions, you may run
   \`oc-search '<specific-term>'\` for targeted session history lookups.

6. Write a brief internal summary of:
   - The purpose of this change
   - How it fits into recent work on this repo
   - Any relevant conventions, patterns, or prior decisions

## Phase 2: Review

1. Read the diff: \`gh pr diff ${pr.number} --repo ${pr.repo}\`
2. Explore surrounding code in the repo for context as needed.
3. Check that tests exist and are appropriate for the change.
4. Judge on:
   - **Correctness:** bugs, logic errors, edge cases
   - **Design:** does this fit the repo's architecture and conventions?
   - **Safety:** security, data handling, error handling
   - **Continuity:** does this align with the recent direction of the codebase?

## Phase 3: Submit with Discipline

Submit your review using \`gh pr review ${pr.number} --repo ${pr.repo}\`.

${autoApproveNote}
${sensitiveNote}

Rules:
- Do NOT comment on style unless it violates an explicit repo rule or a
  repeated historical pattern you found in the context packet.
- Do NOT speculate about intent if the linked issue or history is missing.
  Ask the author instead.
- Cap your review to the highest-signal findings. 3-5 substantive comments
  is better than 15 nitpicks.
- If changes are trivial and safe (docs, deps, typos, formatting) AND the
  author is eligible for auto-approve, use --approve with a brief comment.
- If changes are reasonable but non-trivial, use --comment with specific,
  actionable feedback referencing files and lines.
- If there are clear bugs or security issues, use --request-changes with
  specific fixes.
- NEVER auto-approve changes to authentication, authorization, encryption,
  data handling, or files matching sensitive patterns.`;
}
```

**Step 1: Write the orchestrator**

```typescript
import { writeFileSync } from "node:fs";
import { join } from "node:path";
import { loadConfig } from "./config.js";
import { discoverPrs, getPrDetail } from "./discover.js";
import { isDispatched, markDispatched, writeOutcome } from "./state.js";
import { ensureRepo, createWorktree } from "./worktree.js";
import { buildContextPacket } from "./context.js";
import { buildReviewPrompt } from "./prompt.js";
import { dispatch } from "./dispatch.js";
import { log } from "./log.js";
import type { OutcomeRecord } from "./types.js";

async function main(): Promise<void> {
  const config = loadConfig();
  log(`Starting review cycle (org: ${config.org})`);

  // 1. Discover PRs
  let prs;
  try {
    prs = await discoverPrs(config);
  } catch (err) {
    log(`Failed to discover PRs: ${err}`);
    process.exitCode = 1;
    return;
  }

  log(`Found ${prs.length} open PRs requesting review`);

  // 2. Filter already-dispatched
  const newPrs = prs.filter((pr) => !isDispatched(config.stateDir, pr.repo, pr.number));
  log(`${newPrs.length} new PRs (${prs.length - newPrs.length} already dispatched)`);

  if (newPrs.length === 0) {
    log("Nothing to review. Done.");
    return;
  }

  // 3. Process each PR
  let dispatched = 0;
  let failed = 0;

  for (const pr of newPrs) {
    log(`Processing ${pr.repo}#${pr.number}: ${pr.title} (by @${pr.author})`);

    try {
      // Get PR details
      const detail = await getPrDetail(pr);
      log(`  +${detail.additions} -${detail.deletions} across ${detail.changedFiles} files`);

      // Ensure repo is cloned
      const repoPath = await ensureRepo(config.projectsDir, pr.repo);

      // Create worktree
      const worktreePath = await createWorktree(repoPath, pr.number, detail.headRefName);
      if (!worktreePath) {
        log(`  Skipping: could not create worktree`);
        failed++;
        continue;
      }

      // Build context packet and write to worktree
      log(`  Building context packet...`);
      const contextPacket = await buildContextPacket(pr, detail, config, worktreePath);
      const contextPath = join(worktreePath, ".lgtm-context.md");
      writeFileSync(contextPath, contextPacket);
      log(`  Context packet: ${contextPacket.length} chars`);

      // Build prompt and dispatch
      const prompt = buildReviewPrompt(pr, config);
      await dispatch(worktreePath, prompt);

      // Mark as dispatched and write outcome
      markDispatched(config.stateDir, pr.repo, pr.number);
      const outcome: OutcomeRecord = {
        repo: pr.repo,
        prNumber: pr.number,
        title: pr.title,
        author: pr.author,
        dispatchedAt: new Date().toISOString(),
        contextPacketChars: contextPacket.length,
      };
      writeOutcome(config.stateDir, outcome);

      dispatched++;
      log(`  Dispatched ${pr.repo}#${pr.number}`);
    } catch (err) {
      log(`  Failed: ${err}`);
      failed++;
    }
  }

  log(`Review cycle complete. Dispatched: ${dispatched}, Failed: ${failed}`);
}

main().catch((err) => {
  log(`Fatal error: ${err}`);
  process.exitCode = 1;
});
```
